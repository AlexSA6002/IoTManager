// Это демо конфигурация показывает, как связать компонетны между собой (см.сценарий)

// Кнопка №1 управляет состоянием кнопок №2 и №3
button 1 na Включить#все Свет 1 0 
// Кнопка №2 управляет pwm №1, №2, №3
button 2 na Лампа Свет 2 0
// Кнопка №3 управляет состоянием pin 2 состояние канала инвертировано
button 3 2 Кухня Свет 3 0 1
// pwm канал №1 подключен к PIN 12 начальное состояние 100%
pwm {id:1,name:"Зеленый",pin:12,state:100,page:"Лампа",order:4}
// pwm канал №2 подключен к GPIO 13 начальное состояние 50%
pwm {id:2,name:"Cиний",pin:13,state:50,page:"Лампа",order:5}
// pwm канал №3 подключен к GPIO 15 начальное состояние 0%
pwm {id:3,name:"Красный",pin:15,state:0,page:"Лампа",order:6}

// Показания adc на GPIO 17
// Диапазон измерений от 0 до 1023 привести к значениям от 1 до 100
analog adc 17 Аналоговый#вход Датчики fillgauge 1 1023 1 100 1
